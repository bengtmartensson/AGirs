58 static void readOneDecode() {
59  decode[0] = '\0';
60 #ifdef RECEIVELED
61  LedLcdManager::setLogicLed(receiveled, LedLcdManager::on);
62 #endif
63  irReceiver->enable();
64  // Wait until something arrives
65  while (!irReceiver->isReady())
66  LedLcdManager::checkTurnoff();
67 
68 #ifdef RECEIVELED
69  LedLcdManager::setLogicLed(receiveled, LedLcdManager::off);
70 #endif
71  MultiDecoder multiDecoder(*irReceiver);
72 #ifdef LCD
73  if (multiDecoder.getType() > MultiDecoder::noise) {
74  LedLcdManager::lcdPrint(multiDecoder.getType() == MultiDecoder::nec_ditto
75  ? "." : multiDecoder.getDecode(),
76  multiDecoder.getType() != MultiDecoder::nec_ditto);
77  if (multiDecoder.getType() == MultiDecoder::nec)
78  LedLcdManager::lcdSetCursor(0, 1); // prepare for dittos
79  }
80 #endif
81  MultiDecoder::Type type = multiDecoder.getType();
82  strncpy(decode, multiDecoder.getDecode(), 100);
83 
84  LedLcdManager::setLogicLed(GirsUtils::decode2logicalLed(type), LedLcdManager::blink);
85  irReceiver->disable();
86 }

150 void loop() {
151  LedLcdManager::checkTurnoff();

154 #endif
155  readOneDecode();

157 #ifdef ETHERNET
