211 static void decodeOrDump(IrReader *irReader, Stream& stream) {
212 #ifdef DECODER
213  MultiDecoder multiDecoder(*irReader);
214 #ifdef LCD
215  if (multiDecoder.getType() > MultiDecoder::noise) {
216  LedLcdManager::lcdPrint(multiDecoder.getType() == MultiDecoder::nec_ditto
217  ? "." : multiDecoder.getDecode(),
218  multiDecoder.getType() != MultiDecoder::nec_ditto);
219  if (multiDecoder.getType() == MultiDecoder::nec)
220  LedLcdManager::lcdSetCursor(0, 1); // prepare for dittos
221  }
222 #endif
223 
224 #ifdef DECODELED
225  LedLcdManager::setLogicLed(DECODELED(multiDecoder.getType()), LedLcdManager::blink);
226 #endif
227 #endif
228 
229 #if defined(DECODER) & ! defined(DONT_REPORT_DECODES) // lircd does its own decoding
230  switch (multiDecoder.getType()) {
231  case MultiDecoder::noise:
232  // ignore
233  break;
234  case MultiDecoder::undecoded:
235  irReader->dump(stream); // report data of undecoded signals
236  break;
237  default:
238  stream.println(multiDecoder.getDecode()); // also for timeout
239  break;
240  }
241 #else // ! (defined(DECODER) & ! defined(DONT_REPORT_DECODES))
242  if (irReader->isEmpty())
243  stream.println(F(timeoutString));
244  else
245  irReader->dump(stream);
246 #endif // !DECODER
247 }

249 static bool receive(Stream& stream) {
250  IrReceiverSampler *irReceiver = IrReceiverSampler::getInstance();
251  if (irReceiver == nullptr)
252  irReceiver = IrReceiverSampler::newIrReceiverSampler(captureSize,
253  GirsUtils::receiverPin(receiverNo), GirsUtils::receiverPullup(receiverNo));
254  if (irReceiver == nullptr)
255  return false;
256  irReceiver->setEndingTimeout(receiveEndingTimeout);
257  irReceiver->setBeginningTimeout(beginTimeout);
258  irReceiver->setMarkExcess(IRRECEIVER_MARK_EXCESS);
259  flushIn(stream);
260 #ifdef RECEIVELED
261  LedLcdManager::setLogicLed(receiveled, LedLcdManager::on);
262 #endif
263  irReceiver->enable();
264 #ifdef ARDUINO
265  while (!irReceiver->isReady() && stream.available() == 0)
266 #else
267  std::cout << "** Simulating timeout **" << std::endl;
268  delay(beginTimeout);
269 #endif
270  LedLcdManager::checkTurnoff();
271  bool ready = irReceiver->isReady();
272  irReceiver->disable();
273 #ifdef RECEIVELED
274  LedLcdManager::setLogicLed(receiveled, LedLcdManager::off);
275 #endif
276  if (ready)
277  decodeOrDump(irReceiver, stream);
278  IrReceiverSampler::deleteInstance();
279  return true;
280 }

285 static bool capture(Stream& stream) {
286  IrWidget *irWidget = IrWidgetAggregating::newIrWidgetAggregating(captureSize,
287  GirsUtils::sensorPullup(sensorNo));
288  if (irWidget == nullptr)
289  stream.println(F("This cannot happen"));
290  irWidget->setEndingTimeout(captureEndingTimeout);
291  irWidget->setBeginningTimeout(beginTimeout);
292  irWidget->setMarkExcess(IRSENSOR_MARK_EXCESS);
293  irWidget->reset();
294 #ifdef CAPTURELED
295  LedLcdManager::setLogicLed(captureled, LedLcdManager::on);
296 #endif
297  flushIn(stream);
298 #ifdef ARDUINO
299  irWidget->capture();
300 #else
301  std::cout << "** Simulating timeout **" << std::endl;
302  delay(beginTimeout);
303 #endif
304 
305 #ifdef CAPTURELED
306  LedLcdManager::setLogicLed(captureled, LedLcdManager::off);
307 #endif
308  if (!irWidget->isEmpty()) {
309  // Trying to decode the capture does not make sense,
310  // that is what "receive" is for.
311  irWidget->dump(stream);
312  } else
313  stream.println(F(timeoutString));
314  IrWidgetAggregating::deleteInstance();
315  return true;
316 }

324 static bool sendNamedCommand(Stream& stream, String& remoteName, String& commandName, unsigned int noSends) {
325  const IrNamedRemote* remote = remoteSet.getIrNamedRemote(remoteName.c_str());
326  if (remote == nullptr) {
327  stream.println(F("No such remote"));
328  return false;
329  }
330 
331  const IrNamedCommand* command = remote->getIrNamedCommand(commandName.c_str());
332  if (command == nullptr) {
333  stream.println(F("No such command"));
334  return false;
335  }
336  const IrSignal *irSignal = command->getIrSignal();
337  bool status = sendIrSignal(*irSignal, noSends); // waits, blinks
338  delete irSignal;
339  return status;

518 static bool processCommand(const String& line, Stream& stream) {
519 #ifdef ETHERNET
520  bool quit = false;
521 #endif
522  Tokenizer tokenizer(line);
523  String cmd = tokenizer.getToken();
524 
525  // Decode command
526  if (cmd.length() == 0) {
527  // empty command, do nothing
528  stream.println(F(okString));
529  } else
530 
531 #ifdef CAPTURE
532  if (cmd[0] == 'a' || cmd[0] == 'c') {
533  capture(stream);
534  } else
535 #endif // CAPTURE
536 
537 #ifdef LCD
538  if (isPrefix(cmd, F("lcd"))) { //LCD
539  String rest = tokenizer.getRest();
540  LedLcdManager::lcdPrint(rest);
541  stream.println(F(okString));
542  } else
543 #endif // LCD
544 
545 #ifdef LED
546  if (isPrefix(cmd, F("led"))) {
547  pin_t no = (pin_t) tokenizer.getInt();
548  String value = tokenizer.getToken();
549  bool success = LedLcdManager::setLogicLed(no, value.c_str());
550  stream.println(success ? F(okString) : F(errorString));
551  } else
552 #endif // LED
553 
554  if (isPrefix(cmd, F("modules"))) {
555  stream.println(F(modulesSupported));
556  } else
557 
558 #ifdef FREEMEM
559  if (isPrefix(cmd, F("memory"))) {
560  stream.println(GirsUtils::freeRam());
561  } else
562 #endif
563 
564 #ifdef INFO
565  if (isPrefix(cmd, F("info"))) {
566  info(stream);
567  } else
568 #endif
569 
570 #ifdef PARAMETERS
571  if (cmd[0] == 'p') { // parameter
572  String variableName = tokenizer.getToken();
573  long value = tokenizer.getInt();
574  unsigned long *variable32 = nullptr;
575  uint16_t *variable16 = nullptr;
576  uint8_t *variable8 = nullptr;
577 #if defined(RECEIVE) || defined(CAPTURE)
578  if (isPrefix(F("beg"), variableName))
579  variable32 = &beginTimeout;
580  else
581 #endif
582 #ifdef CAPTURE
583  if (isPrefix(F("capturee"), variableName))
584  variable32 = &captureEndingTimeout;
585 #endif
586 #ifdef RECEIVE
587  if (isPrefix(F("receivee"), variableName))
588  variable32 = &receiveEndingTimeout;
589  else
590 #endif
591 #if defined(RECEIVE) & defined(IRRECEIVER_2_PIN)
592  if (isPrefix(variableName, F("receiver")))
593  variable8 = &receiverNo;
594  else
595 #endif
596 #ifdef CAPTURE
597  if (isPrefix(F("captures"), variableName)) {
598  // TODO: check evenness of value
599  variable16 = &captureSize;
600  } else
601 #endif
602 #ifdef LED
603 #ifdef CONFIGURABLE_LEDS
604 #ifdef TRANSMITLED
605  if (isPrefix(F("transmitl"), variableName))
606  variable8 = &transmitled;
607  else
608 #endif
609 #ifdef CAPTURELED
610  if (isPrefix(F("capturel"), variableName))
611  variable8 = &captureled;
612  else
613 #endif
614 #ifdef RECEIVELED
615  if (isPrefix(F("receivel"), variableName))
616  variable8 = &receiveled;
617  else
618 #endif
619 #ifdef COMMANDLED
620  if (isPrefix(F("commandl"), variableName))
621  variable8 = &commandled;
622  else
623 #endif
624 #endif
625  // TODO: handle blinkTime. This currently means either break encapsulation
626  // of LedLcdManager, or write a nasty special case :-~
627 #endif
628  {
629  }
630 
631  if (variable32 != nullptr) {
632  if (value != Tokenizer::invalid)
633  *variable32 = value;
634 
635  GirsUtils::printVariable(stream, variableName.c_str(), *variable32);
636  } else if (variable16 != nullptr) {
637  if (value != Tokenizer::invalid)
638  *variable16 = static_cast<uint16_t>(value);
639 
640  GirsUtils::printVariable(stream, variableName.c_str(), *variable16);
641  } else if (variable8 != nullptr) {
642  if (value != Tokenizer::invalid)
643  *variable8 = (uint8_t) value;
644 
645  GirsUtils::printVariable(stream, variableName.c_str(), *variable8);
646  } else
647  stream.println(F("No such variable"));
648  } else
649 #endif // PARAMETERS
650 
651 #ifdef ETHERNET
652  if (cmd[0] == 'q') { // quit
653  quit = true;
654  } else
655 #endif
656 
657 #ifdef RECEIVE
658  // TODO: option for force decoding off
659  if (isPrefix(cmd, F("receive"))) { // receive
660  bool status = receive(stream);
661  if (!status)
662  stream.println(F(errorString));
663  } else
664 #endif // RECEIVE
665 
666 #ifdef NAMED_COMMANDS
667  if (cmd[0] == 'n') {
668  uint16_t noSends = static_cast<uint16_t>(tokenizer.getInt());
669  String remoteName = tokenizer.getToken();
670  String commandName = tokenizer.getToken();
671  bool success = sendNamedCommand(stream, remoteName, commandName, noSends);
672  if (success)
673  stream.println(okString);
674  } else
675 
676  if (isPrefix(cmd, "remote")) {
677  String name = tokenizer.getToken();
678  dumpRemote(stream, name);
679  } else
680 #endif
681 
682 #ifdef RESET
683  if (isPrefix(cmd, "reset")) {
684  reset = true;
685  } else
686 #endif
687 
688 #ifdef TRANSMIT
689  if (cmd[0] == 's') { // send
690  // TODO: handle unparsable data gracefully
691  uint16_t noSends = static_cast<uint16_t>(tokenizer.getInt());
692  frequency_t frequency = tokenizer.getFrequency();
693  uint16_t introLength = static_cast<uint16_t>(tokenizer.getInt());
694  uint16_t repeatLength = static_cast<uint16_t>(tokenizer.getInt());
695  uint16_t endingLength = static_cast<uint16_t>(tokenizer.getInt());
696  microseconds_t *intro = new microseconds_t[introLength];
697  microseconds_t *repeat = new microseconds_t[repeatLength];
698  microseconds_t *ending = new microseconds_t[endingLength];
699  for (uint16_t i = 0; i < introLength; i++)
